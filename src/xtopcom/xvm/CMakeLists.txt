add_compile_options(-static -Wall -Werror)

if (XBUILD_TEST)
    add_definitions(-DXENABLE_TESTS)
endif()

if (DISABLE_REAL_STAKE)

endif()

    add_definitions(-DSTATIC_CONSENSUS)
endif()

    add_definitions(-DELECT_WHEREAFTER)
